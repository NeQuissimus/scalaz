Monad

A monad is the combination of an Applicative with a Bind

A monad instance needs to satisfy the monad laws in addition to those defined by Applicative and Bind:

  • Left identity: pure(a).flatMap(f) === f(a)
  • Right identity: x.flatMap(pure) === x
  • Associativity: x.flatMap(f).flatMap(g) === m.flatMap(x => f(x).flatMap(g))

Typical imports

import scalaz.Scalaz._

Instance declaration

scala> import scalaz.Prelude._
import scalaz.Prelude._

scala> import scalaz.ct.MonadClass
import scalaz.ct.MonadClass

scala> val listMonad: Monad[List] = instanceOf(new MonadClass[List] {
     |   def pure[A](a: A): List[A] = List(a)
     |   def ap[A, B](fa: List[A])(f: List[A => B]): List[B] = fa.zip(f).map(t => t._2(t._1))
     |   def flatMap[A, B](ma: List[A])(f: A => List[B]): List[B] = ma.flatMap(f)
     |   def flatten[A](ma: List[List[A]]): List[A] = ma.flatten
     |   def map[A, B](ma: List[A])(f: A => B): List[B] = ma.map(f)
     | })
listMonad: scalaz.Scalaz.Monad[List] = $anon$1@766b6d30

Usage

scala> val l = List(1, 2, 3).flatMap(x => List(x + 5))
l: List[Int] = List(6, 7, 8)

scala> val k = List(2, 3, 4)
k: List[Int] = List(2, 3, 4)

scala> for {
     |   x <- l
     |   y <- k
     | } yield x * y
res0: List[Int] = List(12, 18, 24, 14, 21, 28, 16, 24, 32)
